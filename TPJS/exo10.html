function calculMoyenne() { const notesInput = document.getElementById('note').value; if (notesInput.trim() === '') { alert('entrer vos note :'); return; } const notesArray = notesInput.split(',').map(note => parseFloat(note)); const total = notesArray.reduce((sum, note) => sum + note, 0); const average = total / notesArray.length; const moyenneField = document.getElementById('moyenne'); moyenneField.value = average.toFixed(2); }
